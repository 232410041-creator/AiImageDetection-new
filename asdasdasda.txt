<script>
const MODEL_URL = "https://teachablemachine.withgoogle.com/models/HlT5Nf46a/";

let model;

async function loadModel() {
    try {
        model = await tmImage.load(
            MODEL_URL + "model.json",
            MODEL_URL + "metadata.json"
        );
        console.log("Model loaded");
    } catch (e) {
        alert("Gagal load model");
        console.error(e);
    }
}
loadModel();

async function predictImage(event) {
    if (!model) {
        alert("Model belum siap");
        return;
    }

    const img = document.getElementById("preview");
    img.src = URL.createObjectURL(event.target.files[0]);
    img.style.display = "block";

    img.onload = async () => {
        const predictions = await model.predict(img);

        const best = predictions.reduce((a, b) =>
            a.probability > b.probability ? a : b
        );

        document.getElementById("result").innerHTML =
            `<strong>${best.className}</strong><br>
             ${(best.probability * 100).toFixed(2)}% confidence`;
    };
}
</script>
